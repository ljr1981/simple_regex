note
	description: "Output streams that base64-encode a stream"
	remark: "According to RFC 2045, line breaks should be normalized to CRLF before encoding, and encoded lines should not be longer then 76 characters. It appears that neither of these operations is always done in practice, accordingly they have been made both BOOLEAN options in the creation procedure. Set both options to True for 100%% conformance to RFC 2045."
	library: "Gobo Eiffel Utility Library"
	copyright: "Copyright (c) 2005-2019, Colin Adams and others"
	license: "MIT License"
	date: "$Date$"
	revision: "$Revision$"

class interface
	UT_BASE64_ENCODING_OUTPUT_STREAM

create 
	make

feature -- Status report

	is_line_breaking: BOOLEAN
			-- Are line breaks to be written out after 76 characters?

	is_normalizing: BOOLEAN
			-- Are line-breaks normalized to CRLF before encoding?

	is_closable: BOOLEAN
			-- Can current output stream be closed?
	
feature -- Output

	put_character (c: CHARACTER_8)
			-- Write c to output stream.

	put_string (a_string: READABLE_STRING_8)
			-- Write a_string to output stream.
	
feature -- Basic operations

	close
			-- Try to close output stream if it is closable. Set
			-- `is_open_write` to false if operation was successful.
	
feature -- Constants

	Full_line_count: INTEGER_32 = 76
			-- Maximum line length

	Shift_2_bits: INTEGER_32 = 4

	Shift_4_bits: INTEGER_32 = 16

	Shift_6_bits: INTEGER_32 = 64

	Padding: INTEGER_32 = 64
			-- Pad character ('=')

	Base64_alphabet: ARRAY [CHARACTER_8]
			-- Base64 alphabet
		ensure
			instance_free: class
	
invariant
	triplet_not_void: triplet /= Void
	triplet_capacity: triplet.count = 3
	triplet_count: triplet_count >= 0 and then triplet_count <= 3

end -- class UT_BASE64_ENCODING_OUTPUT_STREAM
