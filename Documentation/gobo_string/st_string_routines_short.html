	split_string (s: STRING_8; c: CHARACTER_8): DS_LINKED_LIST [STRING_8]

		ensure

		require
			str_not_void: str /= Void
		ensure
			str_not_void: str /= Void

	array_to_string (a: ARRAY [STRING_8]): STRING_8
			-- Concatenate all array items into one string;
			-- Return a new string every time.
		require
			a_not_void: a /= Void
			a_count_bigger_zero: a.count > 0
